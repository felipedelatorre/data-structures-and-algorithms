







let findMaleAndFemale = (data) => {
  return data.reduce((acc, cv) => {
    if(cv.gender === 'male' || cv.gender === 'female') acc.push(cv.name);
    return acc;
  }, []).join(' and ');
};


let findShortest = (data) => {
  return data.reduce((acc, cv) => {
    if (acc.height < cv.height) acc = cv;
    return acc;
  }).name;
};
